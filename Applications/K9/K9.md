# K9

A command synthesizer application which searches for `*.k9.json` files in `\K9\Defaults` and recursively under `\Projects\` to build an action list.

The launched commands are run non-elevated with numerous instance environment variables set.

## Example Default (defaults.k9.json)

```json
{
  "actions": [
    {
      "verb": "generate",
      "command": "GenerateProjectFiles.bat",
      "arguments": "",
      "workingDirectory": "{ROOT}",
      "description": "Generate project & solution files.",
      "platform": "Win64"
    },
    {
      "verb": "open",
      "actions": [
        {
          "verb": "vscode",
          "command": "{LOCAL}\\Programs\\Microsoft VS Code\\Code.exe",
          "arguments": "{ROOT}",
          "workingDirectory": "{LOCAL}\\Programs\\Microsoft VS Code\\",
          "description": "Open the workspace in VSCode",
          "platform": "Win64"
        }
      ]
    },
    {
      "verb": "ue",
      "actions": [
        {
          "verb": "editor",
          "actions": [
            {
              "verb": "build",
              "command": "RunUAT.bat",
              "arguments": "BuildGraph -Script=\"Engine/Build/Graph/Examples/BuildEditorAndTools.xml\" -Target=\"Tag Output Files\" -noP4",
              "workingDirectory": "{ROOT}\\Engine\\Build\\BatchFiles",
              "description": "Make a local build of the UE editor for distribution.",
              "platform": "Win64"
            },
            {
              "verb": "stage",
              "command": "RunUAT.bat",
              "arguments": "BuildGraph -Script=\"Engine/Build/Graph/Examples/BuildEditorAndTools.xml\" -Target=\"Copy To Staging Directory\" -noP4",
              "workingDirectory": "{ROOT}\\Engine\\Build\\BatchFiles",
              "description": "Copy the local UE editor build into the staging environment.",
              "platform": "Win64"
            }
          ]
        }
      ]
    },
    {
      "verb": "horde",
      "actions": [
        {
          "verb": "agent",

          "arguments": "BuildGraph -Script=\"Engine/Source/Programs/Horde/BuildHorde.xml\" -Target=\"Build Horde Agent Installer\" -noP4",
          "workingDirectory": "{ROOT}\\Engine\\Build\\BatchFiles",

          "platform": "Win64"
        },
        {
          "verb": "server",

          "workingDirectory": "{ROOT}\\Engine\\Build\\BatchFiles",
          "description": "Build and bundle a docker image of the Horde Server.",
          "platform": "Win64"
        }
      ]
    },
    {
      "verb": "ugs",

        },
        {
          "verb": "installer",
          "command": "RunUAT.bat",
          "arguments": "BuildGraph -Script=Engine/Source/Programs/UnrealGameSync/BuildUGS.xml -Target=\"Build UnrealGameSync Installer\"",
          "workingDirectory": "{ROOT}\\Engine\\Build\\BatchFiles",
          "description": "Make an installer for UnrealGameSync.",
          "platform": "Win64"
        }
      ]
    }
  ]
}
```
